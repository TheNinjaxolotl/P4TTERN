<script>

    let seq = [], playerSeq = [], level = 0, best = 0, gems = 0, locked = true;

    let balance = 0;


    function loadBalance() {

        const saved = localStorage.getItem('totalGems');

        balance = saved ? parseInt(saved) : 0;

        document.getElementById('balance-val').textContent = balance;

        document.getElementById('shop-balance-val').textContent = balance;

    }


    function saveBalance() {

        localStorage.setItem('totalGems', balance);

    }


    function getTile(i) { return document.getElementById('t' + i); }

    function allTiles() { return [0, 1, 2, 3].map(getTile); }


    async function flash(idx) {

        const t = getTile(idx);

        t.classList.add('lit');

        await new Promise(r => setTimeout(r, 450));

        t.classList.remove('lit');

    }


    async function playSequence() {

        locked = true;

        allTiles().forEach(t => t.classList.add('dim'));

        document.getElementById('status').textContent = 'UPLOADING DATA...';

        for (const step of seq) {

            await new Promise(r => setTimeout(r, 250));

            await flash(step);

        }

        document.getElementById('status').textContent = 'REPEAT PATTERN';

        allTiles().forEach(t => t.classList.remove('dim'));

        playerSeq = [];

        locked = false;

    }


    function startGame() {

        loadBalance();

        seq = []; level = 0; gems = 0;

        document.getElementById('gem-count').textContent = '0';


        // Show game UI

        document.getElementById('tile-grid').classList.remove('hidden');

        document.getElementById('status').classList.remove('hidden');

        document.getElementById('gem-row').classList.remove('hidden');


        // Hide home buttons

        document.getElementById('home-buttons').classList.add('hidden');


        document.querySelector('.wrapper').classList.add('playing');

        nextRound();

    }


    function nextRound() {

        level++;

        document.getElementById('level-val').textContent = level;

        seq.push(Math.floor(Math.random() * 4));

        playSequence();

    }


    function handleTap(idx) {

        if (locked) return;

        const t = getTile(idx);

        t.classList.add('lit');

        setTimeout(() => t.classList.remove('lit'), 200);

        playerSeq.push(idx);

        if (playerSeq[playerSeq.length - 1] !== seq[playerSeq.length - 1]) {

            gameOver();

            return;

        }

        gems += 10;

        document.getElementById('gem-count').textContent = gems;

        if (playerSeq.length === seq.length) {

            locked = true;

            document.getElementById('status').textContent = 'SUCCESS';

            setTimeout(nextRound, 800);

        }

    }


    function gameOver() {

        locked = true;

        allTiles().forEach(t => t.classList.add('wrong'));

        if (level > best) best = level;

        document.getElementById('best-val').textContent = best;

        balance += gems;

        saveBalance();

        document.getElementById('balance-val').textContent = balance;

        document.getElementById('shop-balance-val').textContent = balance;

        setTimeout(() => {
